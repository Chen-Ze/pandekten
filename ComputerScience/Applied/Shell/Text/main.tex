\section{grep}

\begin{minted}[mathescape]{shell}

\end{minted}

\section{cut}

\begin{minted}[mathescape]{shell}
# any number of spaces as delimiter
tr -s ' ' < file | cut -d ' ' -f 8
\end{minted}

\section{sed}

\begin{minted}[mathescape]{shell}
# replace 'o' with '0' and output to the terminal
sed 's/o/0/g' input.file
\end{minted}
